    前言

    ​ 由于安服项目中经常要写漏扫报告，我这里写漏扫报告一般都是基于绿盟和Nessus的；不同公司出报告出法都是不一样的，问了些朋友，他们都是直接出原始报告即可（QAQ），但是我还需要在分析下原始报告结果，筛除可利用的漏洞并出一个分析报告（不知道有没有和我一样的，如果有，那你得仔细看了），但是这种ctrl+c，ctrl+v重复性的工作必然会浪费很多时间，于是就有了以下工具来处理此类问题，帮助安服hxdm节省时间

    PS：工具很简单，因为是自己用，大部分参数写死了；可能有些大佬用不上，但还是希望大佬们给一些代码优化上的建议

    判断依据

    ​ 既然要分析，那肯定得有个自己判断的依据，这里采用exploit-DB和Coresecurity网站可利用的漏洞做对比，通过CVE编号做判断，当然你也可以用别的库，比如MSF的库等等；个别漏洞需要根据自己的经验来判断，比如各种未授权访问，这些漏洞都是没有CVE编号的，但是Nessus有插件ID可以做判断，绿盟的可以通过标题做判断，你可以把这些漏洞单独选出来可以做一个人工库。

    实现过程

    ​ 根据自己的word模板往里批量写内容，我们是这样的模板，重复性的工作就是表格内容和下面的漏洞介绍，以前都是ctrl+c,然后ctrl+v，浪费了不少学习时间，当时菜是原罪，不过现在也菜QAQ。

    先看下绿盟的HTML扫描结果和Nessus的CSV扫描结果，这里绿盟的XLS的结果没有漏洞介绍和修复建议，所以就爬了下html页面，提取需要的内容。

    绿盟中爬取了标题，影响IP，漏洞详情，解决方案，CVE，NSFOCUS这些字段

    Nessus直接用导出的CSV做对比即可

    漏洞库：Exploit-DB和Coresecurity（漏洞库不做分享，各位大佬可以自己做个人工库或者自己爬QAQ）

    ​ 先看下程序的几个函数名，整个过程与逻辑都很简单，这个小工具我也是边学边写边问的，可能有些地方优化不好QAQ,在底下源码中我会在注释中介绍每个函数，让刚入门python的小白能看懂，

    源码

    # -*- coding:utf-8 -*-
import docx
from docx import Document
import pandas as pd
from lxml import etree
import csv
html = etree.parse('./index.html', etree.HTMLParser())#绿盟原始html报告做爬取
def match(result, db):#与库中的CVE做对比并筛选出整行
    scan_result = pd.read_csv(result, encoding='utf-8')  # 读取漏扫结果
    db_result = pd.read_csv(db, encoding='utf-8')  # 读取db库
    index = scan_result['CVE'].isin(db_result['CVE']) #基于CVE判断是否存在/可以基于标题或者插件ID
    outfile = scan_result[index] # index为布尔值，这里是取出布尔值为true的那行
    outfile = outfile.dropna(axis=0,subset=['CVE'], how='any')#去除每行的空值，库里有些漏洞是没有CVE的，这里做去重
    return outfile
def html_Parse(level):  #通过颜色判断高中低
    if level == 'color:#E42B00':
        info = '高'
    elif level == 'color:#AF6100':
        info = '中'
    elif level == 'color:#737373':
        info = '低'
    return info
def concat(scan_result, db_file_name, db2_file_name):#这里是做了多个库对比后的合并，这里是对比2个
    outfile = match(scan_result, db_file_name)
    outfile2 = match(scan_result, db2_file_name)
    file = [outfile, outfile2]
    outfile = pd.concat(file)
    return outfile
def merge():#nessus的CSV结果一个漏洞对应一个ip，所以csv中就有很多行，这里将它多行合为1行，一个漏洞对应多个IP
    df = pd.read_csv('Nessus.csv', encoding='gbk')
    csv_data = df.values.tolist()  # 变为集合做去重
    result_dict = &#123;&#125;
    result_set = set()
    # 取出CVE
    for i in csv_data:
        cve = i[1]
        result_set.add(cve)
    cve_ip = &#123;&#125;
    for cve in result_set:
        ip_list = []
        for i in csv_data:
            if i[1] == cve:
                ip_list.append(i[4])
                result_dict[cve] = i
        cve_ip[cve] = ip_list
    res = []
    for i in result_dict:
        ip = ''
        for j in cve_ip[i]:
            j = j + '\n'
            ip += j
        result_dict[i][4] = ip
        res.append(result_dict[i])
    return res#整个过程都是操作的list
def WriteNessus():#写入word模板
    df = merge() #取出nessus的结果
    doc = docx.Document('Template.docx')#获取模板内容
    tbs = doc.tables
    tb = tbs[1]#选择第二个表格
    data_list = []
    for i in range(len(df)):
        Name = df[i][7]
        Host = df[i][4]
        Risk = df[i][3]
        data_list.append([str(i + 1), Name, Risk, Host, ''])#data_list写入需要内容
    for index, i in enumerate(data_list):
        index += 1#我模板标头是有格式的，这里我从第二行开始写
        tb.add_row()#添加表格行
        cells = tb.rows[index].cells  # 第几行的格子
        for col in range(5):  # 设定每个格子的内容
            cells[col].text = str(i[col])
    data_list = []
    for i in range(len(df)):
        Name = df[i][7]
        Description = df[i][9]
        Host = df[i][4]
        Solution = df[i][10]
        data_list.append([Name, Description, Host, Solution])#data_list写入需要的内容
    for i in range(len(df)):#写入漏洞介绍，IP，建议
        doc.add_paragraph(data_list[i][0], style='Heading 3')
        p1 = doc.add_paragraph()
        text1 = p1.add_run("漏洞介绍")
        doc.add_paragraph(data_list[i][1])
        p2 = doc.add_paragraph()
        text2 = p2.add_run("涉及资产IP")
        doc.add_paragraph(data_list[i][2])
        p3 = doc.add_paragraph()
        text3 = p3.add_run("修复建议")
        doc.add_paragraph(data_list[i][3])
        text1.bold = True
        text2.bold = True
        text3.bold = True
    doc.save("Nessus_result.docx")
def WriteCSV(num):#将绿盟html结果中需要的信息写入csv，方便做提取
    with open('lv.csv', 'a', newline='', encoding='utf-8') as csvfile:
        writer = csv.writer(csvfile)
        writer.writerow(['Title', 'Content', 'Level', 'IP_result', 'Solution', 'CVE', 'NSFOCUS'])
        for i in range(num):
            i += 1
            title = html.xpath(
                '/html/body/div/div[4]/div[8]/div[2]/table/tbody/tr[' + str(2 * i - 1) + ']/td[2]/span/text()')
            level = html.xpath(
                '/html/body/div/div[4]/div[8]/div[2]/table/tbody/tr[' + str(2 * i - 1) + ']/td[2]/span/@style')
            content = html.xpath(' //*[@id="table_4_1_' + str(i) + '"]/td/table/tr[2]/td/text()')
            solution = html.xpath('//*[@id="table_4_1_' + str(i) + '"]/td/table/tr[3]/td/text()')
            CVE = html.xpath('//*[@id="table_4_1_' + str(i) + '"]/td/table/tr[7]/td/a/text()')
            NSFOCUS = html.xpath('//*[@id="table_4_1_' + str(i) + '"]/td/table/tr[9]/td/a/text()')
            IP = html.xpath('//*[@id="table_4_1_' + str(i) + '"]/td/table/tr[1]/td/a[last()]/@href')
            level = ('').join(level).strip()
            level = html_Parse(level)
            IP = ('').join(IP)
            html_IP = etree.parse(IP, etree.HTMLParser())
            IP_tr = html_IP.xpath('//*[@id="report"]/div[3]/div/table/tbody/tr')
            IP_result = ''
            for k in range(len(IP_tr)):
                k += 1
                ip = html_IP.xpath('//*[@id="report"]/div[3]/div/table/tbody/tr[' + str(k) + ']/td[2]/a/text()')
                xx = ip[0].strip()
                IP_result += xx + '\n'
            title = ('').join(title).strip()
            content = ('').join(content).strip()
            solution = ('').join(solution).strip()
            CVE = ('').join(CVE).strip()
            NSFOCUS = ('').join(NSFOCUS).strip()
            writer.writerow([title, content, level, IP_result, solution, CVE, NSFOCUS])
    csvfile.close()
